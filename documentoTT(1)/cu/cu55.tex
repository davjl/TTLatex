\begin{UseCase}{CU55}{Aceptar voucher de pago}{
		Registra en el sistema que el voucher de pago es aprobado.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{De Jesús López David.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Admin. caja.}
		\UCitem{Propósito}{Cambiar el estado del voucher de pago como aceptado.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{\MSGref{MSG100-}{Voucher aceptado}.}
		\UCitem{Precondiciones}{Tener un voucher de pago.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{\BRref{RN20}{Validaciones de voucher}.}
		\UCitem{Errores}{\MSGref{MS101-}{No pudo cambiarse el estado del voucher}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en \IUbutton{Aceptar} en la pantalla \IUref{IU54}{Ver voucher de pago}. \label{CU54Aceptar}
		\UCpaso Sube el archivo en el sistema. \Trayref{A}
		\UCpaso Muestra en la pantalla \IUref{IU54}{Ver voucher de pago} el mensaje {\bf MSG100-} ''Voucher aceptado''.
		\UCpaso Redirecciona a la pantalla \IUref{IU49}{Gestionar pagos caja}. 
	\end{UCtrayectoria}
	
	\begin{UCtrayectoriaA}{A}{El sistema no se encuentra disponible}
		\UCpaso Muestra el Mensaje {\bf MSG101-} ''No pudo cambiarse el estado del voucher'' en la pantalla \IUref{IU54}{Ver voucher de pago} 
		\UCpaso[\UCactor] Continua en \ref{CU54Aceptar} del \UCref{CU55}.
	\end{UCtrayectoriaA}
