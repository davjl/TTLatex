\begin{UseCase}{CU14}{Cargar comprobante de pago}{
		Permite al usuario adjuntar una imagen o archivo pdf que representa su comprobante de pago para su posterior revisión mediante la pantalla \IUref{IU14}{Cargar Comprobante de Pago}.
		}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{ Roberto Mendoza Saavedra.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Usuario: Alumno, Empleado, Externo.}
		\UCitem{Propósito}{Tener un medio para adjuntar un comprobante de pago en formato pdf o imagen en formato válido.}
		\UCitem{Entradas}{Archivo con formato válido.}
		\UCitem{Salidas}{\MSGref{MSG2}{Se adjunto el archivo correctamente}, \MSGref{MSG3}{Archivo enviado correctamente}}
		\UCitem{Precondiciones}{Ingresar un archivo con formato válido.}
		\UCitem{Postcondiciones}{
			\begin{itemize}
				\item Se cargó un archivo en el sistema.
				\item El archivo cambia de estado con base en la regla de negocio \BRref{RN03}{Estado del archivo}.
			\end{itemize}}
		\UCitem{Reglas de Negocio}{\BRref{RN03}{Estado del archivo}.}
		\UCitem{Errores}{\MSGref{MSG56}{Formato de archivo no válido}, \MSGref{MSG57}{Operación no completada}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso Muestra la pantalla \IUref{IU14}{Cargar Comprobante de Pago}.
		\UCpaso[\UCactor] Presiona el icono...
		\UCpaso Muestra un selector de archivos.
		\UCpaso[\UCactor] Selecciona un archivo y da clic en el botón \IUbutton{Aceptar} del selector de archivos. 
		\UCpaso Muestra el nombre del archivo en un campo de la pantalla \IUref{IU14}{Cargar Comprobante de Pago}.
		\UCpaso[\UCactor] Presiona el botón \IUbutton{Enviar} de la pantalla \IUref{IU14}{Cargar Comprobante de Pago}.
		\UCpaso Valida si el formato del archivo es el correcto. \Trayref{A}
		\UCpaso Cambia el estado del archivo a PENDIENTE.\Trayref{B}
		\UCpaso Redirige a la pantalla \IUref{IU16}{Gestionar Pagos Usuario}.
	\end{UCtrayectoria}

		\begin{UCtrayectoriaA}{A}{El usuario adjuntó un archivo con formato no válido.}
		\UCpaso Muestra el mensaje \MSGref{MSG56}{Formato de archivo no válido}.
		\UCpaso[\UCactor] Continua en el paso 2 del caso de uso \UCref{CU14}.
	\end{UCtrayectoriaA}

		\begin{UCtrayectoriaA}{B}{El sistema no puede procesar el envío.}
		\UCpaso Muestra el mensaje \MSGref{MSG57}{Operación no completada}.
		\UCpaso[\UCactor] Continua en el paso 4 del caso de uso \UCref{CU14}.
	\end{UCtrayectoriaA}
