\begin{UseCase}{CU5}{Login}{
		Permite al usuario ingresar al sistema mediante un nombre de usuario y contraseña.\\Dependiendo el perfil de identificación se redireccionará a la pantalla \IUref {IU5.1}{Bienvenida} con sus permisos correspondientes.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López .}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Contador, Dentista, Coordinador CELEX, Administrador Fotocopiado, Subdirector Administrativo, Administrador Biblioteca, Cajero, Usuario: Alumno, Empleado, Externo.}
		\UCitem{Propósito}{Garantizar el acceso sólo a usuarios autorizados.}
		\UCitem{Entradas}{Datos del usuario: 
		\begin{itemize}
		    \item Correo electrónico
		    \item Contraseña
		\end{itemize}}
		\UCitem{Salidas}{Ninguna.}
		\UCitem{Precondiciones}{Tener un registro previo en el sistema}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{\MSGref{MSG50}{Campo obligatorio},\MSGref{MSG51}{Campo no válido}, \MSGref{MSG53}{Formato no válido}, \MSGref{MSG55-} {Usuario y/o contraseña incorrectos.}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Ingresa a la pantalla \IUref{IU5}{Login}.%\label{CU17Login}.
		\UCpaso[\UCactor] Ingresa correo electrónico, contraseña y presiona el botón de \IUbutton {Ingresar}
		\UCpaso Verifica que el usuario haya ingresado todos los campos obligatorios. \Trayref{A}
		\UCpaso Verifica que los campos ingresados sean válidos.\Trayref{B}
		\UCpaso Verifica que el formato del correo electrónico sea el correcto. \Trayref{C}
		\UCpaso Verifica que el correo electrónico y la contraseña correspondan.\Trayref{D}
		\UCpaso Obtiene el perfil correspondiente a la cuenta y muestra la pantalla \IUref{IU5.1}{Bienvenida}
	\end{UCtrayectoria}
		
		\begin{UCtrayectoriaA}{A}{El usuario no ingreso un e-mail valido}
			\UCpaso Muestra el Mensaje {\bf MSG60-} ''El e-mail no existe en el sistema'' en la pantalla \IUref{IU5}{Login} 
			\UCpaso Continua en el paso \ref{CU1login} del \UCref{CU5}.
		\end{UCtrayectoriaA}
		
		\begin{UCtrayectoriaA}{B}{El password es incorrecto}
			\UCpaso  Muestra el Mensaje {\bf MSG61-} ``El password es incorrecto.'' en la pantalla  \IUref{IU5}{Login} 
			\UCpaso Continua en el paso \ref{CU1login} del \UCref{CU5}.
		\end{UCtrayectoriaA}
		
		\begin{UCtrayectoriaA}{C}{Se supera el numero de intentos al ingresar contraseña}
			\UCpaso  Muestra el Mensaje {\bf MSG62-} ``Se supero el numero de intentos permitidos.'' en la pantalla \IUref{IU5}{Login} 
			\UCpaso[] Fin del caso de uso
		\end{UCtrayectoriaA}
