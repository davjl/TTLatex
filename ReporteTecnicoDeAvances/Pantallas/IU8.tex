\subsection{IU8 Visualizar servicios}

\subsubsection{Objetivo}{
	Mostrar las diferentes áreas de pago y poder seleccionar una de interes.
}
	
\subsubsection{Diseño}
	Se mostrará información basica del usuario y un menú desplegable con las áreas de pago, además del botón \IUbutton{Mostrar}.

\IUfig[.6]{gui/Usuario/VisualizarServicios}{IU8}{Visualizar servicios}

%\subsubsection{Salidas}

%	Comentarios ingresados debajo de la información del componente electrónico.

%\subsubsection{Entradas}


\subsubsection{Comandos}
\begin{itemize}
	\item Menú desplegable: El menú desplegable servirá para que el usuario pueda visualizar y seleccionar el área de interes en consultar los servicios disponibles.
	\item \IUbutton{Mostrar}: Dependiendo de la selección en el menú desplegable se rediccionará a las pantallas:
	\begin{itemize}
		\item \IUref{IU9}{Visualizar servicios dentales}
		\item \IUref{IU10}{Visualizar servicios CELEX}
		\item \IUref{IU11}{Visualizar servicios fotocopiado}
		\item \IUref{IU12}{Visualizar servicios biblioteca}
	\end{itemize}
\end{itemize}
