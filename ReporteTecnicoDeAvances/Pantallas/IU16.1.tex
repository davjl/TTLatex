\subsection{IU16.1 Visualizar de voucher pago}

\subsubsection{Objetivo}{
	Permite al contador y subdirector administrativo visualizar el voucher de pago.
}
	
\subsubsection{Diseño}
	Se muestra el voucher de pago que se envió a caja y un botón \IUbutton{Regresar}.

\IUfig[.6]{gui/Contador/Voucher}{IU16.1}{Visualizar voucher contador y subdirector administrativo}

	\item \IUbutton{Regresar}: Según el actor se redirecciona a la pantalla \IUref{IU38}{Gestionar pagos contaduría} o \IUref{IU55}{Gestionar pagos subdirección}.
\end{itemize}
