\subsection{IU23 Gestionar citas}

\subsubsection{Objetivo}{
	Mostrar una lista con todos los usuarios que han realizado una cita para consulta. 
}
	
\subsubsection{Diseño}
	Se muestra una caja de texto que sirve como buscador, una tabla con los campos: \begin{itemize}
		\item Estatus paciente.
		\item Nombre.
		\item Curp.
		\item Número de boleta o empleado.
		\item Fecha.
		\item Hora.
		\item Acciones.
	\end{itemize} En el apartado de acciones se utilizan los iconos 
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/cuestionario}}},
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/consulta}}},
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/historialClinico}}},
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/reagenda}}} y 
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/noAsistio}}}.
	Además de una descripción del significado de colores para el status del paciente.

\IUfig[.6]{gui/Dentales/GestionarCitas}{IU23}{Gestionar citas}

\subsubsection{Comandos}
\begin{itemize}
	\item 
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/cuestionario}}}: Redirecciona a la pantalla \IUref{IU26}{Visualizar cuestionario}.
	\item 
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/consulta}}}: Redirecciona a la pantalla \IUref{IU27}{Realizar consulta}.
	\item 
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/historialClinico}}}: Redirecciona a la pantalla \IUref{IU30}{Visualizar historial clínico}.
	\item 
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/reagenda}}}: Redirecciona a la pantalla \IUref{IU25}{Reagendar}.
	\item 
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/noAsistio}}}: Cambia el estado del paciente a inasistencia.
\end{itemize}
