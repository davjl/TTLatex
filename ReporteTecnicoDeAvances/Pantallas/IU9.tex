\subsection{IU9 Visualizar servicios dentales}

\subsubsection{Objetivo}{
	Mostrar los diferentes servicios disponibles, precios y una pequeña descripción de estos. 
}
	
\subsubsection{Diseño}
	Se muestra una caja de texto que sirve como buscador, una tabla con los campos:\begin{itemize}
		\item Servicio.
		\item Precio.
		\item Descripción.
	\end{itemize}, además de los botones \IUbutton{Agendar cita} y \IUbutton{Regresar}.

\IUfig[.6]{gui/Usuario/VisualizarServiciosDentales}{IU9}{Visualizar Servicios Dentales}
