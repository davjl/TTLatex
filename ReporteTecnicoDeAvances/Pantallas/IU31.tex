\subsection{IU31 Visualizar consulta.}

\subsubsection{Objetivo}{
	Visualizar una descripción de un tratamiento realizado anteriormente.
}
	
\subsubsection{Diseño}
	Muestra los datos del paciente registrado, una lista de checkbox con los conceptos más comunes de consulta y un panel desplegable con la cantidad en caso de ser posible, dos cajas de texto, un odontograma y un botón \IUbutton{Regresar}.

\IUfig[.6]{gui/Dentales/VisualizarConsulta}{IU31}{Visualizar consulta}

\end{itemize}
