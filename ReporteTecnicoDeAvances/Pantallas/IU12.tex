\subsection{IU12 Visualizar servicios biblioteca}

\subsubsection{Objetivo}{
	Mostrar los diferentes servicios disponibles, precios y una pequeña descripción de estos. 
}
	
\subsubsection{Diseño}
	Se muestra una caja de texto que sirve como buscador, una tabla con los campos:\begin{itemize}
		\item Servicios.
		\item Precio.
		\item Descripción.
		\item Acciones.
	\end{itemize}, Dentro de acciones se muestran los siguientes iconos 
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/spei}}},
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/subirArchivo}}}
	, y un botón \IUbutton{Regresar}.

\IUfig[.6]{gui/Usuario/VisualizarServiciosBiblioteca}{IU12}{Visualizar Servicios biblioteca}

\subsubsection{Comandos}
\begin{itemize}
	\item \raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/subirArchivo}}}: Redirecciona a la pantalla de \IUref{IU14}{Cargar comprobante de pago}
	\item
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/spei}}}: Redirecciona a la pantalla de \IUref{IU13}{Pago electrónico}
	\item \IUbutton{Regresar}: Redirecciona a la pantalla \IUref{IU8}{Visualizr servicios}
\end{itemize}
