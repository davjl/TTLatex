\subsection{IU16 Visualizar comprobante de pago}

\subsubsection{Objetivo}{
	Visualizar el comprobante de pago enviado a caja.
}
	
\subsubsection{Diseño}
	Se muestra el comprobante de pago que se envió a caja para revisión, un botón \IUbutton{Regresar} y en caso de que el comprobante este en un estado de rechazo se mostrará la descripción del motivo.

\IUfig[.6]{gui/Usuario/VisualizarComprobantePago}{IU16}{Visualizar comprobante de pago}

%\subsubsection{Entradas}


\subsubsection{Comandos}
\begin{itemize}
	\item \IUbutton{Regresar}: Redirecciona a la pantalla \IUref{IU15}{Gestionar pagos usuario}.
\end{itemize}
