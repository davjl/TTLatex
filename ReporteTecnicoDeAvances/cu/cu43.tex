\begin{UseCase}{CU43}{Aceptar voucher de pago}{
		Registra en el sistema que el voucher de pago es aprobado.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Administrador Caja.}
		\UCitem{Propósito}{Cambiar el estado del voucher de pago como aceptado.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{\MSGref{MSG5-}{Comprobante aceptado}.}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{\BRref{RN20}{Validaciones de voucher}.}
		\UCitem{Errores}{\MSGref{MSG59-}{No se pudo cambiar el estado del comprobante}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en el botón \IUbutton{Aceptar} en la pantalla \IUref{IU16}{Visualizar comprobante pago}.
		\UCpaso Sube el archivo en el sistema. \Trayref{A}
		\UCpaso Muestra en la pantalla \IUref{IU16}{Visualizar comprobante pago} el mensaje {\bf MSG5-} ''Comprobante aceptado''.
		\UCpaso Redirecciona a la pantalla \IUref{IU15}{Gestionar pagos}. 
	\end{UCtrayectoria}
	
	\begin{UCtrayectoriaA}{A}{El sistema no se encuentra disponible}
		\UCpaso Muestra el Mensaje {\bf MSG59-} ''No pudo cambiarse el estado del comprobante'' en la pantalla \IUref{IU16}{Visualizar comprobante pago} 
		\UCpaso[\UCactor] Continua en 1 del \UCref{CU43}.
	\end{UCtrayectoriaA}
