\begin{UseCase}{CU16}{Marcar inasistencia}{
		Permite al dentista marcar una falta al usuario y liberar la hora de cita agendada.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{7 de septiembre de 2018.}
		\UCitem{Actor}{Usuario.}
		\UCitem{Propósito}{Mostrar como accesible la hora apartada de la cita.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Hora accesible.}
		\UCitem{Precondiciones}{Haber una cita agendada.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguno.}
		\UCitem{Tipo}{Secundario.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en el botón \IUbutton {Citas} del menú principal de la pantalla \IUref{IU5.4}{Bienvenida área dental} 
		\UCpaso Muestra la pantalla \IUref{IU23}{Gestionar citas}.
		\UCpaso Muestra las opciones de realizar consulta, historial clínico, marcar inasistencia y cancelar cita.
		\UCpaso[\UCactor] Selecciona una cita agendada.
	\end{UCtrayectoria}
	
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en la opción realizar consulta.\\
	{\bf Región de la trayectoria:}Paso 4 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU19}.\\		
	
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en la opción historial clínico.\\
	{\bf Región de la trayectoria:}Paso 4 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU22}.\\
	
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en la opción marcar inasistencia.\\
	{\bf Región de la trayectoria:}Paso 4 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU16}.\\
	
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en la opción cancelar cita.\\
	{\bf Región de la trayectoria:}Paso 4 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU17}.\\
