\begin{UseCase}{CU9}{Visualizar servicios dentales}{
		Permite al Usuario visualizar el catálogo de servicios y precios disponibles para el área de odontología mediante la pantalla \IUref {IU09}{Visualizar servicios dentales}.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Usuario: Alumno, Empleado, Externo.}
		\UCitem{Propósito}{Tener un medio de visualización del catálogo de  servicios disponibles en el área dental.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Tabla que muestra las columnas de:
			\begin{itemize}
				\item Concepto.
				\item Precio.
				\item Descripción.
			\end{itemize}.}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguno}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
			\UCpaso[\UCactor] Presiona el botón \IUbutton{Mostrar} de la pantalla \IUref{IU08}{Visualizar Servicios}.
			\UCpaso Obtiene el catálogo de servicios disponibles con su concepto, precio y descripción para el área dental.
			\UCpaso Muestra la pantalla \IUref{IU9}{Visualizar servicios dentales}.
	\end{UCtrayectoria}
