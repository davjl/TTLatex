\begin{UseCase}{CU7}{Visualizar Notificaciones}{
		Proporciona al cajero, subdirector administrativo, áreas de servicio y usuario un medio de información capaz de informarle sobre los cambios de estado en la gestión de pagos dentro del sistema.\\
}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{César Erick Hernández López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{10 de abril de 2018.}
		\UCitem{Actor}{Usuario, Cajero, Subdirector Administrativo, Coordinador CELEX, Administrador Fotocopiado, Dentista y Administrador Biblioteca}
		\UCitem{Propósito}{Informar de cambios en los estados de pago de interes para cada área u otras actividades que necesiten de una rapida notificación}
		\UCitem{Entradas}{Ninguna}
		\UCitem{Salidas}{Ninguna}
		\UCitem{Precondiciones}{
			\begin{itemize}
				\item Haber sido realizada una acción de prioridad que necesite ser notificada inmediatamente
			\end{itemize}
}
		\UCitem{Postcondiciones}{Ninguna}
		\UCitem{Reglas de Negocio}{\BRref{RN8}{Acciones que requieren notificación}.}
		\UCitem{Errores}{Ninguna}
		\UCitem{Tipo}{Primario}
	\end{UseCase}

		\UCpaso Redirecciona a la pantalla \IUref{IU7}{Notificaciones}.
		\UCpaso Con base en la regla de negocio \BRref{RN8}{Acciones que requieren notificación}  despliega un listado de las acciones a notificar.
	\end{UCtrayectoria}
