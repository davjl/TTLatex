\begin{UseCase}{CU53}{Guardar comprobante sigaa}{
		Agrega en sistema el archivo comprobante sigaa, validando los pagos recibidos.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{De Jesús López David.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Admin. caja.}
		\UCitem{Propósito}{Tener el registro en sistema el comprobante sigaa.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{\MSGref{MSG90-}{Archivo guardado correctamente}.}
		\UCitem{Precondiciones}{Tener el archivo de comprobante sigaa en espera de guardado}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{\MSGref{MSG91-}{Error al guardar}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en \IUbutton{Enviar} en la pantalla \IUref{IU51}{Adjuntar pago sigaa}.\label{CU53Guardar}
		\UCpaso Sube el archivo en el sistema. \Trayref{A} 
		\UCpaso Muestra en la pantalla \IUref{IU51}{Adjuntar pago sigaa} el mensaje {\bf MSG80-} ''Archivo guardado correctamente''.
	\end{UCtrayectoria}
	
	\begin{UCtrayectoriaA}{A}{El sistema no se encuentra disponible}
		\UCpaso Muestra el Mensaje {\bf MSG91-} ''El archivo no pudo guardarse'' en la pantalla \IUref{IU51}{Adjuntar pago sigaa} 
		\UCpaso[\UCactor] Continua en \ref{CU53Guardar} del \UCref{CU53}.
	\end{UCtrayectoriaA}
