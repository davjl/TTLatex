\begin{UseCase}{CU39}{Gestionar pagos caja}{
		Permite al Cajero visualizar los comprobantes que han sido mandados por el Usuario para realizar el pago de servicios.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Cajero.}
		\UCitem{Propósito}{Otorgar un medio para la visualización de todos los comprobantes de pagos del día.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Tabla que muestra las columnas de:
			\begin{itemize}
				\item Estatus.
				\item Nombre.
				\item CURP.
				\item Boleta.
				\item Fecha de entrega.
				\item Cantidad
				\item Concepto
				\item Acciones
			\end{itemize}.}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguna.}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en el botón \IUbutton{Gestionar pagos} del menu \IUref{IU5.1.1}{Menú principal}.
		\UCpaso Muestra la pantalla \IUref{IU44}{Gestionar pagos } con los iconos 
		\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/verPAgo}}},
		\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/archivoSigaa}}} y el botón \IUbutton{Imprimir todos}.
		
	\end{UCtrayectoria}

	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en el icono
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/verPago}}} para visualizar el comprobante de pago\\
	{\bf Región de la trayectoria:}Paso 2 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU16}.\\\\
	{\bf Causa de la extensión:}El actor da clic en el icono
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/archivoSigaa}}} para visualizar el comprobante sigaa\\
	{\bf Región de la trayectoria:}Paso 2 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU32}.\\\\
	{\bf Causa de la extensión:}El actor da clic en botón \IUbutton {Imprimir todos}\\
	{\bf Región de la trayectoria:}Paso 2 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU37}.\\\\
