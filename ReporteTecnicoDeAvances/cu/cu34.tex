\begin{UseCase}{CU34}{Visualizar archivo de pagos por año}{
		Permite a los actores visualizar un histórico de pagos aprobados por caja ordenados por año mediante la pantalla \IUref{IU34}{Archivo de pagos}. Todos los pagos mostrados ya habrán sido aprobados y archivados por el cajero.
		}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{Roberto Mendoza Saavedra.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Contador, Subdirector Administrativo, áreas de servicio.}
		\UCitem{Propósito}{Otorgar una pantalla de visualización para selección de un año especifico en el rango de los últimos cinco años.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Tabla que muestra las columnas de:
			\begin{itemize}
				\item Año.
				\item Total ingresos.
				\item Acciones.
			\end{itemize}.}
		\UCitem{Precondiciones}{Deben de existir pagos aprobados por el Cajero.}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{\BRref{RN14}{Almacenamiento de archivos}.}
		\UCitem{Errores}{Ninguno.}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Selecciona la opción {\bf Archivos de pago} del {\bf Menú principal} en las pantallas  \IUref{IU5.2}{Bienvenida contador}, \IUref{IU5.3}{Bienvenida cajero}, \IUref{IU5.4}{Bienvenida área dental},
		\IUref{IU5.5}{Bienvenida CELEX}.
		\UCpaso Muestra la pantalla \IUref{IU34}{Archivo de Pagos por año}.
		\UCpaso Con base en la regla de negocio \BRref{RN14}{Almacenamiento de archivos} muestra los registros de los ultimos cinco años.
		\UCpaso Muestra los registros con el icono
		\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/ver}}}
	\end{UCtrayectoria}
		
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en el icono
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/ver}}}
	para visualizar los pagos por mes del año seleccionado.\\
	{\bf Región de la trayectoria:}Paso 3 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU35}.\\
%-------------------------------------- TERMINA descripción del caso de uso.
