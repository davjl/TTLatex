		Permite al dentista cancelar una cita por otros motivos.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{7 de septiembre de 2018.}
		\UCitem{Actor}{Usuario.}
		\UCitem{Propósito}{Termiinar la cita sin haber efectuado.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Cita cancelada.}
		\UCitem{Precondiciones}{Haber una cita agendada.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguno.}
		\UCitem{Tipo}{Secundario.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en la opción cancelar cita de la pantalla \IUref{IU23}{Gestionar citas} 
		\UCpaso Cancela la cita.
		\UCpaso Se notifica al usuario.
		\UCpaso Muestra la pantalla \IUref{IU23}{Gestionar citas} sin la cita que acaba de cancelar.
	\end{UCtrayectoria}
