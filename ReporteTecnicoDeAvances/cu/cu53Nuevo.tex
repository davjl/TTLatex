\begin{UseCase}{CU53}{Visualizar pagos por fecha}{
		Permite al Contador, Subdirector Administrativo, Dentista, Coordinadora Celex, Administrador Fotocopiado, Administrador Biblioteca visualizar los pagos que ya fueron aceptados por caja en un orden por día.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Contador, Subdirector Administrativo, Dentista, Coordinadora Celex, Administrador Fotocopiado, Administrador Biblioteca.}
		\UCitem{Propósito}{Visualizar los pagos aceptados en caja por orden de día.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Tabla que muestra las columnas de:
			\begin{itemize}
				\item Nombre.
				\item Curp.
				\item Boleta.
				\item Fecha.
				\item Cantidad.
				\item Acciones.
			\end{itemize}.}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguno.}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Selecciona el icono 
		\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/ver}}}
		en la pantalla \IUref{IU35}{Visualizar archivo de pagos por mes}
		\UCpaso Muestra la pantalla \IUref{IU36}{Pagos por fecha}.
		\UCpaso Muestra los registros , el botón \IUbutton {Imprimir todos} y el icono \raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/verPago}}} 
	\end{UCtrayectoria}
	
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en el icono
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/verPago}}} para visualizar el comprobante sigaa.\\
	{\bf Región de la trayectoria:}Paso 3 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU30}.\\\\
	{\bf Causa de la extensión:}El actor da clic en el botón \IUbutton {Imprimir todos}.\\
	{\bf Región de la trayectoria:}Paso 3 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU27}.\\
