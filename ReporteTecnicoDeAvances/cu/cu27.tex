\begin{UseCase}{CU27}{Gestionar pagos contaduría}{
		Permite al Contador visualizar los pagos del día confirmados por el Cajero.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Contador.}
		\UCitem{Propósito}{Otorgar un medio para la visualización de todos los pagos del día confirmados por el Cajero.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Tabla que muestra las columnas de:
			\begin{itemize}
				\item Servico.
				\item Precio.
				\item Fecha de entrega.
				\item Acciones
			\end{itemize}.}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguna.}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en el botón \IUbutton{Gestionar pagos} en el \IUref{IU5.1.1}{Menú principal}.
		\UCpaso Muestra la pantalla \IUref{IU27}{Gestionar pagos contaduría} con los iconos 
		\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/verPago}}},
		\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/archivoSigaa}}} y el botón \IUbutton{Imprimir todos}
		
	\end{UCtrayectoria}

	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en el icono
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/verPago}}} para visualizar el pago\\
	{\bf Región de la trayectoria:}Paso 2 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU9}.\\\\
	
	{\bf Causa de la extensión:}El actor da clic en el icono
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/archivoSigaa}}} para visualizar el pago sigaa\\
	{\bf Región de la trayectoria:}Paso 2 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU23}.\\\\
	
	{\bf Causa de la extensión:}El actor da clic en botón \IUbutton {Imprimir todos}\\
	{\bf Región de la trayectoria:}Paso 2 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU26}.\\\\
