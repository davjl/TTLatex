\begin{UseCase}{CU15}{Enviar comprobante de pago}{
		Permite subir al sistema el comprobante de pago seleccionado.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{De Jesús López David.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Usuario: Alumno, Empleado, Externo.}
		\UCitem{Propósito}{Subir el archivo de pago al sistema para ser visible en área de caja.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{\MSGref{MSG70-}{Archivo enviado correctamente}.}
		\UCitem{Precondiciones}{Tener el archivo de comprobante en espera de envió}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{\MSGref{MSG71-}{El archivo no pudo enviarse}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en \IUbutton{Enviar} en la pantalla \IUref{IU14}{Cargar comprobante pago}. \label{CU2Cargar}
		\UCpaso Sube el archivo en el sistema. \Trayref{A}
		\UCpaso Muestra en la pantalla \IUref{IU14}{Cargar comprobante pago} el mensaje {\bf MSG70-} ''Archivo enviado correctamente''.
	\end{UCtrayectoria}
	
	\begin{UCtrayectoriaA}{A}{El sistema no se encuentra disponible}
		\UCpaso Muestra el Mensaje {\bf MSG71-} ''El archivo no pudo enviarse'' en la pantalla \IUref{IU14}{Cargar comprobante de pago} 
		\UCpaso[\UCactor] Continua en \ref{CU2Cargar} del \UCref{CU15}.
	\end{UCtrayectoriaA}
