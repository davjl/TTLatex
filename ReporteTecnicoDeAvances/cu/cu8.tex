\begin{UseCase}{CU8}{Visualizar servicios}{
		Permite al Usuario visualizar una lista de los servicios disponibles en el sistema para la gestión de pagos mediante la pantalla \IUref{IU8}{Visualizar servicios}.\\\\Estos servicios disponibles serán los de Biblioteca, CELEX, Dentales y Fotocopiado.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{Roberto Mendoza Saavedra.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{17 de abril de 2018.}
		\UCitem{Actor}{Usuario.}
		\UCitem{Propósito}{Tener un medio de visualización de todos los servicios disponibles para la gestión de pagos.}
		\UCitem{Entradas}{Servicio seleccionado para la gestión de pagos.}
		\UCitem{Salidas}{
			\begin{itemize}
				\item Identificador.
				\item Lista de servicios disponibles para la gestión de pagos.
			\end{itemize}.}
		\UCitem{Precondiciones}{Debe de existir por lo menos un servicio disponible para la gestión de pagos.}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguno.}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
			\UCpaso[\UCactor]Selecciona la opción {\bf Visualizar servicios} del {\bf Menú principal} en la pantalla \IUref{IU5.1}{Bienvenida usuarios.}
			\UCpaso Obtiene la lista de todos los servicios disponibles para la gestión de pagos.
			\UCpaso Muestra al Usuario la pantalla \IUref{IU8}{Visualizar servicios}.
			\UCpaso[\UCactor]Selecciona un servicio de la lista para realizar su gestión de pagos.
			%\UCpaso Valida que el Usuario haya ingresado el campo marcado como obligatorio en la pantalla \IUref{IU08}{Visualizar Servicios}.\Trayref{A}			
	\end{UCtrayectoria}
