\begin{UseCase}{CU40}{Cargar pagos}{
		Permite al Cajero guardar en el sistema todos los pagos aceptados del día y preservar los registros de pagos rechazados o pendientes de aprobación.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Admin. caja.}
		\UCitem{Propósito}{Otorgar un medio para guardar en el sistema los pagos aceptados del día y conservar los rechazos o los pendientes de aprobación.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{\MSGref {MSG4-}{Archivos cargados correctamente}}
		\UCitem{Precondiciones}{Debe existir al menos un pago aceptado.}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{\BRref{RN04}{Fecha limite de pago}.}
		\UCitem{Errores}{\MSGref{MSG58-}{No se pudo cargar los pagos}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en el botón \IUbutton{Cargar pagos} en la pantalla \IUref{IU39}{Gestionar pagos caja}.
		\UCpaso Elimina los pagos no realizados o rechazados.
		\UCpaso Guarda los pagos aceptados en el sistema y genera un reporte de pagos. \Trayref{A}
		\UCpaso Muestra mensaje {\bf MSG4-} ''Archivos cargados correctamente''.
	\end{UCtrayectoria}
			
	\begin{UCtrayectoriaA}{A}{El sistema no se encuentra disponible}
		\UCpaso Muestra el Mensaje {\bf MSG58-} ''No se pudo cargar los pagos'' en la pantalla \IUref{IU39}{Gestionar pagos caja} 
		\UCpaso[\UCactor] Continua en 1 del \UCref{CU40}.
	\end{UCtrayectoriaA}
