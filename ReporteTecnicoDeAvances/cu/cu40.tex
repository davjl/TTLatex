\begin{UseCase}{CU40}{Cargar pagos}{
		Permite al Cajero guardar en el sistema todos los pagos aceptados del día, preservar los pendientes de aprobación y eliminar los rechazados.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Admin. caja.}
		\UCitem{Propósito}{Otorgar un medio para guardar en el sistema los pagos aceptados del día y conservar  los pendientes de aprobación.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{\MSGref {MSG4}{Archivos cargados correctamente}}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{\BRref{RN20}{Horarios corte de caja}.}
		\UCitem{Errores}{\MSGref{MSG59}{No se pudo cargar el pago}, \MSGref{MSG62}{La hora de corte de caja a terminado}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Basado en la regla de negocio \BRref{RN20}{Horarios corte de caja} Da clic en el botón \IUbutton{Cargar pagos} en la pantalla \IUref{IU39}{Gestionar pagos caja}. \Trayref{A}
		\UCpaso Elimina los pagos no rechazados.
		\UCpaso Mantiene en pantalla los pagos pendientes de revisión.
		\UCpaso Guarda los pagos aceptados en el sistema y genera un reporte de pagos. \Trayref{B}
		\UCpaso Muestra mensaje \MSGref {MSG4}{Archivos cargados correctamente}.
	\end{UCtrayectoria}
	
	\begin{UCtrayectoriaA}{A}{El corte de caja es fuera del horario}
		\UCpaso Muestra el Mensaje \MSGref{MSG62}{La hora de corte de caja a terminado} en la pantalla \IUref{IU39}{Gestionar pagos caja} 
		\UCpaso[\UCactor] Continua en 1 del \UCref{CU40}.
	\end{UCtrayectoriaA}
			
	\begin{UCtrayectoriaA}{B}{El sistema no se encuentra disponible}
		\UCpaso Muestra el Mensaje \MSGref{MSG59}{No se pudo cargar el pago} en la pantalla \IUref{IU39}{Gestionar pagos caja} 
		\UCpaso[\UCactor] Continua en 1 del \UCref{CU40}.
	\end{UCtrayectoriaA}
