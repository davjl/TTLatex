\begin{UseCase}{CU40}{Cargar pagos}{
		Permite realizar un corte de caja con los pagos aceptados y eliminar los registros de pago rechazados o sin efectuar.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{De Jesús López David.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Admin. caja.}
		\UCitem{Propósito}{Crear un reporte de pagos aceptados por el personal en turno.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Reporte de corte de caja, \MSGref {MSG4-}{Archivos cargados correctamente}}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{\MSGref{MSG58-}{No se pudo cargar los pagos}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en el botón \IUbutton{Cargar pagos} en la pantalla \IUref{IU15}{Gestionar pagos}.
		\UCpaso Muestra mensaje {\bf MSG4-} ''Archivos cargados correctamente'' la pantalla \IUref{IU38}{Pagos por fecha}.
	\end{UCtrayectoria}
			
	\begin{UCtrayectoriaA}{A}{El sistema no se encuentra disponible}
		\UCpaso Muestra el Mensaje {\bf MSG58-} ''No se pudo cargar los pagos'' en la pantalla \IUref{IU49}{Gestionar pagos} 
		\UCpaso[\UCactor] Continua en 1 del \UCref{CU40}.
	\end{UCtrayectoriaA}
