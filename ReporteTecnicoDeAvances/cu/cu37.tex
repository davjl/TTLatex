		\UCitem{Salidas}{Comprobantes impresos \MSGref{MSG1-}{Operación exitosa}}
		\UCitem{Precondiciones}{Tener el registro minimo de un pago.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguna.}
		\UCitem{Tipo}{Primario.}
