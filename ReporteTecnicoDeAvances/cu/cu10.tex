\begin{UseCase}{CU10}{Visualizar servicios CELEX}{
		Permite al Usuario visualizar el catálogo de servicios y precios disponibles en el departamento del CELEX mediante la pantalla \IUref {IU10}{Visualizar servicios CELEX}.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{ David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Usuario: Alumno, Empleado, Externo.}
		\UCitem{Propósito}{Tener un medio de visualización del catálogo de servicios disponibles en el área del CELEX, así como otorgar opciones de gestión de pagos.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Ninguna.}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguno}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Presiona el botón \IUbutton{Mostrar} de la pantalla \IUref{IU08}{Visualizar Servicios}.
		\UCpaso Obtiene el catálogo de servicios disponibles con su concepto, precio y descripción para el área de CELEX.
		\UCpaso Muestra al Usuario la pantalla \IUref{IU10}{Visualiar servicios CELEX} con los iconos \raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/subir_archivo}}}
	\end{UCtrayectoria}

	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor requiere realizar el pago del servicio de forma electrónica.\\
	{\bf Región de la trayectoria:}Paso 2 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU13}.\\\\

	{\bf Causa de la extensión:}El actor requiere subir un archivo de pago.\\
