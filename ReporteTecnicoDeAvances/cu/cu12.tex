\begin{UseCase}{CU12}{Visualizar servicios de biblioteca}{
		Permite al Usuario visualizar el catálogo de servicios y precios disponibles para el área de biblioteca mediante la pantalla \IUref {IU12}{Visualizar servicios de biblioteca}.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{Roberto Mendoza Saavedra.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{17 de abril de 2018.}
		\UCitem{Actor}{Usuario: Alumno, Empleado, Externo.}
		\UCitem{Propósito}{Tener un medio de visualización del catálogo de los servicios disponibles en el área de biblioteca, así como otorgar opciones de gestión de pagos.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Tabla que muestra las columnas de:
			\begin{itemize}
				\item Concepto.
				\item Precio.
				\item Descripción.
				\item Acciones.
			\end{itemize}.}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguno}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		
			\UCpaso[\UCactor] Presiona el botón \IUbutton{Mostrar} de la pantalla \IUref{IU08}{Visualizar Servicios}.
			\UCpaso Obtiene el catálogo de servicios disponibles con su concepto, precio y descripción para el área de biblioteca.
			\UCpaso Muestra al Usuario la pantalla \IUref{IU12}{Visualizar servicios de biblioteca} con los iconos
			\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/spei}}}
			\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/subirArchivo}}}
	\end{UCtrayectoria}

	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en el icono
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/spei}}} para realizar el pago del servicio de forma electrónica.\\
	{\bf Región de la trayectoria:}Paso 3 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU13}.\\\\
	%---
	{\bf Causa de la extensión:}El actor da clic en el icono
	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/subirArchivo}}} para subir su comprobante de pago al sistema.\\
	{\bf Región de la trayectoria:}Paso 3 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU14}.
		
%-------------------------------------- TERMINA descripción del caso de uso.
