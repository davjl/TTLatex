\begin{UseCase}{CU12}{Gestionar notas de pago}{
		Permite al usuario tener un control de las notas de pago generadas por las áreas de servicio.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{7 de septiembre de 2018.}
		\UCitem{Actor}{Usuario.}
		\UCitem{Propósito}{Manter un registro de las notas de pago.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Tabla que muestra las columnas de:
			\begin{itemize}
				\item Servicio.
				\item Cantidad.
				\item Acciones.
			\end{itemize}.}
		\UCitem{Precondiciones}{Tener al menos una nota de pago.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguno.}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en el botón \IUbutton {Notas de pago} del menú principal de la pantalla \IUref{IU5.1}{Bienvenida usuarios}.
		\UCpaso Muestra la pantalla \IUref{IU19}{Gestionar notas de pago usuarios}.
		\UCpaso Muestra los registros de notas y las opciones de ver pago, subir archivo de pago y pago en caja.
	\end{UCtrayectoria}
	
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en la opción de ver pago.\\
	{\bf Región de la trayectoria:}Paso 4 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU13}.\\		
	
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en la opción de subir archivo de pago.\\
	{\bf Región de la trayectoria:}Paso 4 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU7}.\\		
	
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El actor da clic en la opción de pago en caja.\\
	{\bf Región de la trayectoria:}Paso 4 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU10}.\\
