\begin{UseCase}{CU5}{Login}{
		Permite al Contador, Dentista, Coordinador CELEX, Administrador Fotocopiado, Subdirector Administrativo, Administrador Biblioteca, Cajero y Usuario ingresar al sistema mediante un nombre de usuario y contraseña.\\Dependiendo el perfil de identificación se redireccionará a la pantalla \IUref {IU5.1}{Bienvenida} con sus permisos correspondientes.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López .}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Contador, Dentista, Coordinador CELEX, Administrador Fotocopiado, Subdirector Administrativo, Administrador Biblioteca, Cajero, Usuario: Alumno, Empleado, Externo.}
		\UCitem{Propósito}{Garantizar el acceso sólo a usuarios autorizados.}
		\UCitem{Entradas}{Datos del usuario: 
		\begin{itemize}
		    \item Correo electrónico
		    \item Contraseña
		\end{itemize}}
		\UCitem{Salidas}{Ninguna.}
		\UCitem{Precondiciones}{Tener un registro previo en el sistema}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{\MSGref{MSG50}{Campo obligatorio},\MSGref{MSG55} {Usuario y/o contraseña incorrectos.}}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}		

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Ingresa a la pantalla \IUref{IU5}{Login} con opciones solo para el actor Usuario de recuperar contraseña y registar usuario.%\label{CU17Login} .
		\UCpaso[\UCactor] Ingresa correo electrónico, contraseña y presiona el botón de \IUbutton {Ingresar}
		\UCpaso Verifica que el usuario haya ingresado todos los campos obligatorios. \Trayref{A}
		\UCpaso Verifica que el correo electrónico y la contraseña correspondan.\Trayref{B}
		\UCpaso Obtiene el perfil correspondiente a la cuenta y muestra la pantalla \IUref{IU5.1}{Bienvenida}
	\end{UCtrayectoria}
		
		\begin{UCtrayectoriaA}{A}{Faltan datos obligatorios}
			\UCpaso Muestra el Mensaje \MSGref{MSG50}{Campo obligatorio} en la pantalla \IUref{IU5}{Login} 
			\UCpaso Continua en el paso 2 del \UCref{CU5}.
		\end{UCtrayectoriaA}

		\begin{UCtrayectoriaA}{B}{Usuario y/o contraseña incorrectos}
			\UCpaso  Muestra el Mensaje \MSGref{MSG55}{Usuario y/o contraseña incorrectos} en la pantalla \IUref{IU5}{Login} 
			\UCpaso Continua en el paso 2 del \UCref{CU5}
		\end{UCtrayectoriaA}
	
	{\bf Puntos de extensión}\\\\
	{\bf Causa de la extensión:}El Usuario selecciona la opción de recuperar contraseña.\\
	{\bf Región de la trayectoria:}Paso 1 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU4}.\\\\
	
	{\bf Causa de la extensión:}El Usuario selecciona la opción de registrar usuario.\\
	{\bf Región de la trayectoria:}Paso 1 de la Trayectoria Principal.\\
	{\bf Extiende a:}\UCref{CU1}.
