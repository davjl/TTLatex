\begin{UseCase}{CU11}{Visualizar historial de pagos}{
		Permite al usuario visualizar su historial de pagos realizados.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{David De Jesús López.}
		\UCitem{Estatus}{Edición.}
		\UCitem{Fecha de último estatus}{9 de abril de 2018.}
		\UCitem{Actor}{Usuario.}
		\UCitem{Propósito}{Visualizar los pagos aceptados en caja por orden de día.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Tabla que muestra las columnas de:
			\begin{itemize}
				\item Fecha.
				\item Servicio.
				\item Cantidad.
				\item Acciones.
			\end{itemize}.}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}	
		\UCitem{Reglas de Negocio}{Ninguna.}
		\UCitem{Errores}{Ninguno.}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Da clic en el botón \IUbutton {Historial de pagos} del menú principal de la pantalla \IUref{IU5.1}{Bienvenida usuarios}.
		\UCpaso Muestra la pantalla \IUref{IU18}{Visualizar historial de pagos}.
		\UCpaso Muestra los registros de pago y el icono \raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/archivoSigaa}}} 
	\end{UCtrayectoria}
	
	{\bf Puntos de extensión}\\\\

	\raisebox{-\mydepth}{\fbox{\includegraphics[height=\myheight]{icons/archivoSigaa}}} para visualizar el comprobante SIG@.\\
	{\bf Región de la trayectoria:}Paso 3 de la Trayectoria Principal.\\
